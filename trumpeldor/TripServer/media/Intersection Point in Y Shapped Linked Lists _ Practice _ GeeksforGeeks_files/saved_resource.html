        Here's a simple trick. As the values are always positive, so traverse through the first linked list, and multiply all values by -1.
        Now start traversing from the second head. If any negative value is encountered, means that is the merge point. So return negative of that value.

        int intersectPoint(Node* head1, Node* head2){
        Node *temp = head1;
        while(temp){
        temp->data = -1 * temp->data;
        temp = temp->next;
        }
        temp = head2;
        int ans = -1;
        while(temp){
        if(temp->data < 0){
        ans = -temp->data;
        break;
        }
        temp = temp->next;
        }
        return ans;
        }

                                                                                                  int intersectPoint(Node* head1, Node* head2)
                                                                                                  {
                                                                                                  struct Block {
                                                                                                  struct Node *node;
                                                                                                  struct Block *next;
                                                                                                  };
                                                                                                  struct Block *arr[1001];
                                                                                                  for(int i = 0; i < 1001; i++)
                                                                                                  arr[i] = NULL;
                                                                                                  struct Node *current = head2;
                                                                                                  struct Block *temp;
                                                                                                  while(current != NULL)
                                                                                                  {
                                                                                                  if(arr[(current->data)] == NULL)
                                                                                                  {
                                                                                                  temp = (struct Block*)malloc(sizeof(struct Block));
                                                                                                  temp->node = current;
                                                                                                  temp->next = NULL;
                                                                                                  arr[current->data] = temp;
                                                                                                  }
                                                                                                  else
                                                                                                  {
                                                                                                  temp = (struct Block*)malloc(sizeof(struct Block));
                                                                                                  temp->node = current;
                                                                                                  temp->next = arr[current->data];
                                                                                                  arr[current->data] = temp;
                                                                                                  }
                                                                                                  current = current->next;
                                                                                                  }
                                                                                                  current = head1;
                                                                                                  bool found = false;
                                                                                                  struct Block *curBlock = NULL;
                                                                                                  while(current!=NULL)
                                                                                                  {
                                                                                                  if(arr[current->data] != NULL)
                                                                                                  {
                                                                                                  curBlock = arr[current->data];
                                                                                                  while(curBlock!=NULL)
                                                                                                  {
                                                                                                  if(curBlock->node == current)
                                                                                                  {
                                                                                                  found = true;
                                                                                                  break;
                                                                                                  }
                                                                                                  else
                                                                                                  {
                                                                                                  curBlock = curBlock->next;
                                                                                                  }
                                                                                                  }
                                                                                                  if(found)
                                                                                                  break;
                                                                                                  }
                                                                                                  current = current->next;
                                                                                                  }
                                                                                                  if(!found)
                                                                                                  return -1;
                                                                                                  else
                                                                                                  return(current->data);
                                                                                                  }

                                                                                                                                            simple C++ code(execution time : 0.01) :

                                                                                                                                            Trick: Multiply all element of first linked list with -1 then search for negative element in second linked list & return it. Trick works since values in both list is always +ve (given: 1<=value<=1000)

                                                                                                                                            int intersectPoint(Node* head1, Node* head2)
                                                                                                                                            {
                                                                                                                                            // Your Code Here
                                                                                                                                            Node* temp1=head1;
                                                                                                                                            while(temp1){
                                                                                                                                            temp1->data*=-1;
                                                                                                                                            temp1=temp1->next;
                                                                                                                                            }
                                                                                                                                            Node* temp2=head2;
                                                                                                                                            while(temp2){
                                                                                                                                            if(temp2->data<0)
                                                                                                                                            return -1*temp2->data;
                                                                                                                                            temp2=temp2->next;
                                                                                                                                            }
